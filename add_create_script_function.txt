def create_executable_script(file_path, script_content, make_executable=True):
    """Create an executable script file
    
    Args:
        file_path: Path where script should be created
        script_content: Content of the script
        make_executable: If True, set executable permissions (Unix/Linux/Mac)
    """
    import stat
    
    with open(file_path, 'w') as f:
        f.write(script_content)
